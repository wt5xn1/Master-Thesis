\newglossaryentry{Stokes Theorem}{
    name={stokes},
    description={Theorem relating the surface integral of a flux 
    to the line integral on the boundary of the surface. For a 
    continuous, smooth vector field $\vb{F}(x,y,z)$ on a surface \Sigma,
    \int\int\limits{\Sigma} (\nabla\times\vb{F})\cdot d\Sigma = 
    \oint\limits{\partial \Sigma} \vf{F}\cdot d(\partial\Sigma)}
