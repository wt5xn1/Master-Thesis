    \sum\limits_{\substack{k=-\infty\\k\neq 0}}^\infty 
    \Cnk[0,k] \In{0} \left( \frac{2\pi}{L}|k|r \right)

\newcommand{\Bffno}{
    \sum\limits_{\substack{n=-\infty\\n\neq 0}}^\infty 
    \Cnk[n,0] r^{|n|}\exp(jn\varphi)
}

    \sum\limits_{\substack{n=-\infty\\n\neq 0}}^\infty 
    \sum\limits_{\substack{k=-\infty\\k\neq 0}}^\infty 
    \Cnk[n,k] j^{-k}\In{|n|} \left( \frac{2\pi}{L}|k|r \right)
    \exp \left( j \frac{2\pi}{L}kz \right)
    \exp(jn\varphi)
}
